@article{campbell2002deep-blue, title = {Deep Blue}, journal = {Artificial Intelligence}, volume = {134}, number = {1}, pages = {57-83}, year = {2002}, issn = {0004-3702}, doi = {https://doi.org/10.1016/S0004-3702(01)00129-1}, url = {https://www.sciencedirect.com/science/article/pii/S0004370201001291}, author = {Murray Campbell and A.Joseph Hoane and Feng-hsiung Hsu}, keywords = {Computer chess, Game tree search, Parallel search, Selective search, Search extensions, Evaluation function}, abstract = {Deep Blue is the chess machine that defeated then-reigning World Chess Champion Garry Kasparov in a six-game match in 1997. There were a number of factors that contributed to this success, including: •a single-chip chess search engine,•a massively parallel system with multiple levels of parallelism,•a strong emphasis on search extensions,•a complex evaluation function, and•effective use of a Grandmaster game database. This paper describes the Deep Blue system, and gives some of the rationale that went into the design decisions behind Deep Blue.} }
